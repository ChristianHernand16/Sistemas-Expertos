# Probabilidad Bayes
def regla_bayes(a, p, b):
# regla P(n)
    n = 1 - a
# regla P(b)
    p_b = p * a + b * n
# regla P(a|b)
    g = (p * a) / p_b
    return g
# P(a)
a = 0.6
# P(b|a)
p = 0.36
# P(b|n)
b = 0.24
# regla P(a|b)
comprobacion = regla_bayes(a, p, b)
#Realizamos la respectiva sumarizacion
print('P(a|b) = %.6f%%' % (comprobacion))
